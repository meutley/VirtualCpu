# Add
#   3 + 7 (D0 = 10)
SET	D0, 3
SET	D1, 7
ADD	D0, D1


# Subtract
#   10 - 5 (D0 = 5)
SET	D1, 5
SUBTRACT D0, D1


# Bitwise OR
#   D0 = 5 (0101)
#   D1 = 2 (0010)
#   D0 OR D1 = 0111 (7)
SET	D1, 2
OR	D0, D1
